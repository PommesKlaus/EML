\subsection{Formeln}

\begin{equation}
    \text{Euklidische Norm} = \sum_{l=1}^{n}|x^i_l-x^j_l|
\end{equation}

\begin{equation}
    \text{Euklidische Distanz} = \|\cdot \|=\sqrt{(x^i_1-x^j_1)^2+\dots +(x^i_n-x^j_n)^2}
\end{equation}

\begin{equation}
    \text{z-Transformation} = \hat{x}^{(i)}_j = \frac{x^{(i)}_j-\bar{x}_j}{\sigma_j}
\end{equation}
